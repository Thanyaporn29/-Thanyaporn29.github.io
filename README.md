# เริ่มต้นพัฒนา Streamlit
1. clone จาก Thanyaporn29
         -git clone

3. deactivate
         -conda deactivate

4. สร้างสภาพแวดล้อมใหม่
         -python -m venv venv

5. activate สภาพแวดล้อมใหม่
         -venv/scripts/activate

6. ติดตั้ง streamlit
         -pip install streamlit
